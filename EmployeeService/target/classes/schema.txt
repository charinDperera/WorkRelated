CREATE TABLE KT_DEPARTMENT(
	DEP_ID INT(22) NOT NULL,
	DEP_NAME VARCHAR(40) NOT NULL,
	PRIMARY KEY(DEP_ID)
	);

CREATE TABLE KT_EMPLOYEE(
	EMP_ID INT(22) NOT NULL,
	DEPARTMENT INT(22) NOT NULL,
	EMP_NAME VARCHAR(40) NOT NULL,
	DOB INT(20) NOT NULL,
	CONSTRAINT fk_kt_department_dep_id FOREIGN KEY (DEPARTMENT) REFERENCES KT_DEPARTMENT(DEP_ID),
	PRIMARY KEY(EMP_ID)
	);
